#include<stdio.h>
int main()
{
int year;
printf("Enter a year:");
scanf("%d", &year);
year=((year%4==0)&& (year%100!=0)||(year%400==0));
switch (year)
{case 1:
printf("%d is a leap year.\n",year);
break;
case 0:
printf("%d is not a leap year.\n",year);
break;
default:
printf("Error in calculation.\n");
}
return 0;
}
