#include <stdio.h>

int main() {
    int matrix[3][3] = {{5, 2, 9}, {1, 7, 3}, {6, 4, 8}};
    int max = matrix[0][0];
    int min = matrix[0][0];

    // Traverse the matrix
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            if (matrix[i][j] > max) {
                max = matrix[i][j];
            }
            if (matrix[i][j] < min) {
                min = matrix[i][j];
            }
        }
    }

    printf("Maximum element: %d\n", max);
    printf("Minimum element: %d\n", min);

    return 0;
}
