#include <stdio.h>

int main() {
    int row, col, i, j, isIdentity = 1;
    int matrix[10][10];

    // Input the number of Rows and Columns
    printf("Enter the number of rows and columns (must be square): ");
    scanf("%d %d", &row, &col);

    // Check if it is a square matrix
    if (row != col) {
        printf("The matrix is not a square matrix, so it cannot be an identity matrix.\\n");
        return 0;
    }

    // Input elements into the matrix
    printf("Enter the elements of the matrix:\\n");
    for (i = 0; i < row; i++) {
        for (j = 0; j < col; j++) {
            printf("Element [%d][%d]: ", i, j);
            scanf("%d", &matrix[i][j]);
        }
    }

    // Check whether the matrix is an identity matrix
    for (i = 0; i < row; i++) {
        for (j = 0; j < col; j++) {
            if (i == j && matrix[i][j] != 1) {
                // Diagonal element should be 1
                isIdentity = 0;
                break;
            } else if (i != j && matrix[i][j] != 0) {
                // Non-diagonal elements should be 0
                isIdentity = 0;
                break;
            }
        }
        if (isIdentity == 0) {
            break; // Exit outer loop if not an identity matrix
        }
    }

    // Display the result
    if (isIdentity == 1) {
        printf("The given matrix is an Identity Matrix.\\n");
    } else {
        printf("The given matrix is not an Identity Matrix.\\n");
    }

    return 0;
}
